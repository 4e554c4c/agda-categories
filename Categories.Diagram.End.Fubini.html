module Categories.Diagram.End.Fubini {o ℓ e o′ ℓ′ e′ o′′ ℓ′′ e′′}
  {C : Category o ℓ e} {D : Category o′ ℓ′ e′} {E : Category o′′ ℓ′′ e′′ }
  (F : Bifunctor (Category.op C ×ᶜ Category.op D) (C ×ᶜ D) E) where
--open Wedges F
module C = Category C
module D = Category D
module F = Functor F

private
  -- just shuffling arguments
  -- TODO give this operation a name?
  F′ : Bifunctor (C.op ×ᶜ C) (D.op ×ᶜ D) E
  F′ = F ∘F ((πˡ ∘F πˡ ※ πˡ ∘F πʳ) ※ (πʳ ∘F πˡ ※ πʳ ∘F πʳ))

  --postulate
  --  F⇒F′ : NaturalTransformation F F′

module _ {eₚ : ∫ F} {eᵢ : ∫ (F′ ♯)} {eₒ : ∫ (∫.E eᵢ)} where
  open M E

  module eₚ = ∫ eₚ
  module eᵢ = ∫ eᵢ
  module eₒ = ∫ eₒ

  open _≅_
  open Iso
  open DinaturalTransformation using (α)
  open NaturalTransformation using (η)
  Fubini : ∫.E eₚ ≅ ∫.E eₒ
  Fubini .to = eₚ.factor record
    { E = ∫.E eₒ
    ; dinatural = dtHelper record
      { α = λ { (X , Y) → (eᵢ.dinatural.α X <∘ eₒ.dinatural) .α Y }
      ; commute = λ { (f , g) → {! !} }
      }
    }
  Fubini .from = eₒ.factor record
    { E = ∫.E eₚ
    ; dinatural = dtHelper record
      { α = λ Y → eᵢ.factor record
        { E = const eₚ.E
        ; dinatural = dtHelper record
          { α = λ X → ntHelper record
            { η = λ { (X' , Y') → {! eₚ.dinatural.α (X , Y)!} }
            ; commute = {! !}
            }
          ; commute = {! !}
          }
        } .η (Y , Y)
      ; commute = {! !}
      }
    }
  Fubini .iso .isoʳ = {! eₚ.universal !}
  Fubini .iso .isoˡ = {! eₚ.universal !}
