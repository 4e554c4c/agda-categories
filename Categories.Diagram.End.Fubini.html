module Categories.Diagram.End.Fubini {o ℓ e o′ ℓ′ e′ o′′ ℓ′′ e′′}
  {C : Category o ℓ e} {D : Category o′ ℓ′ e′} {E : Category o′′ ℓ′′ e′′ }
  (F : Bifunctor (Category.op C ×ᶜ Category.op D) (C ×ᶜ D) E) where
--open Wedges F
module C = Category C
module D = Category D
module E = Category E
open module F = Functor F using (F₀; F₁)

private
  -- just shuffling arguments
  -- TODO give this operation a name?
  F′ : Bifunctor (D.op ×ᶜ D) (C.op ×ᶜ C) E
  F′ = F ∘F ((πˡ ∘F πʳ ※ πˡ ∘F πˡ) ※ (πʳ ∘F πʳ ※ πʳ ∘F πˡ))

  F′′ : Bifunctor (C.op ×ᶜ C) (D.op ×ᶜ D) E
  F′′ = F′ ∘F Swap

module _ {ω : ∀ ((p , q) : D.Obj × D.Obj) → ∫ (appˡ F′ (p , q))} {eᵢ : ∫ (⨏ F′ {ω})} where
  open M E

  private
    ∫F = (⨏ F′ {ω})
    module ∫F = Functor ∫F
    module eᵢ = ∫ eᵢ

    module ω ((p , q) : D.Obj × D.Obj) = ∫ (ω (p , q))

  open _≅_
  open Iso
  open DinaturalTransformation using (α)
  open NaturalTransformation using (η)
  open Wedge renaming (E to apex)
  open Category E
  open import Categories.Morphism.Reasoning E
  open HomReasoning
  -- we show that wedges of (∫.E e₀) are in bijection with wedges of F
  -- following MacLane §IX.8
  private module _ (ρ : Wedge ∫F) where
    private
      open module ρ = Wedge ρ using () renaming (E to x)
    -- first, our wedge gives us a wedge in the produ9ct
    ξ : Wedge F
    ξ .apex = x
    ξ .dinatural = dtHelper record
      { α = λ { (c , p) → ω.dinatural.α (p , p) c ∘ ρ.dinatural.α p }
      ; commute = λ { {c , p} {d , q} (f , s) → begin
          F₁ ((C.id , D.id) , (f , s)) ∘ (ω.dinatural.α (p , p) c ∘ ρ.dinatural.α p) ∘ id
          ≈⟨ {! !} ⟩
          (F₁ ((C.id , D.id) , (C.id , s)) ∘ F₁ ((C.id , D.id) , (f , D.id)) ∘ ω.dinatural.α (p , p) c ∘ id) ∘ ρ.dinatural.α p ∘ E.id
          ≈⟨ (refl⟩∘⟨ ω.dinatural.commute (p , p) f ) ⟩∘⟨refl ⟩
          (F₁ ((C.id , D.id) , (C.id , s)) ∘ F₁ ((f , D.id) , (C.id , D.id)) ∘ ω.dinatural.α (p , p) d ∘ id) ∘ ρ.dinatural.α p ∘ E.id
          ≈⟨ {! !} ⟩
          F₁ ((f , s) , (C.id , D.id)) ∘ (ω.dinatural.α (q , q) d ∘ ρ.dinatural.α q) ∘ id
          ∎
        }
      }

  -- now the opposite direction
  private module _ (ξ : Wedge F) where
    private
      open module ξ = Wedge ξ using () renaming (E to x)
    ρ : Wedge ∫F
    ρ .apex = x
    ρ .dinatural = dtHelper record
      { α = λ p → ω.factor (p , p) record
        { E = x
        ; dinatural = dtHelper record
          { α = λ c → ξ.dinatural.α (c , p)
          ; commute = λ f → ξ.dinatural.commute (f , D.id)
          }
        }
      ; commute = {! !}
      }

  -- This construction is actually stronger than Fubini. It states that the
  -- iterated end _is_ an end in the product category.
  -- Thus the product end need not exist.
  eₚ : ∫ F
  eₚ .∫.wedge = ξ eᵢ.wedge
  eₚ .∫.factor W = eᵢ.factor (ρ W)
  eₚ .∫.universal {W} {c , p} = begin
    (ξ eᵢ.wedge) .dinatural.α (c , p) ∘ eᵢ.factor (ρ W)
      ≈⟨ Equiv.refl ⟩
    (ω.dinatural.α (p , p) c ∘ eᵢ.dinatural.α p)  ∘ eᵢ.factor (ρ W)
      ≈⟨ assoc ⟩
    ω.dinatural.α (p , p) c ∘ (eᵢ.dinatural.α p  ∘ eᵢ.factor (ρ W))
      ≈⟨ refl⟩∘⟨ eᵢ.universal {ρ W} {p} ⟩
    ω.dinatural.α (p , p) c ∘ (ρ W) .dinatural.α p
      ≈⟨ ω.universal (p , p) ⟩
    W .dinatural.α  (c , p)
      ∎
  eₚ .∫.unique {W} {g} h = eᵢ.unique λ { {A = p} → Equiv.sym $ ω.unique (p , p) (sym-assoc ○ h) }

  -- corollary
  Fubini : {eₚ' : ∫ F} → ∫.E eᵢ ≅ ∫.E eₚ'
  Fubini {eₚ'} = end-unique eₚ eₚ'
